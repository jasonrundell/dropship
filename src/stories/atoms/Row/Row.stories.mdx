# Row

## A container used for diving content into rows.

The Row component is used to create a container for spacing around elements. It
can be used to create a row of elements, or to create a container for a single
element.

<ArgsTable story="Default" />

export const Template = (args) => <Row {...args} />

export const Example = () => (
  <>
    <Button label="Button 1" />
    <Button label="Button 2" />
    <Button label="Button 3" />
  </>
)

### Controls

Experiment with this story with the Controls addon in the Canvas tab.

<Canvas>
  <Story
    name="controls"
    args={{
      justify: 'start',
      children: Example()
    }}
    // https://storybook.js.org/docs/react/api/mdx#play-function
    play={async ({ canvasElement }) => {
      const canvas = within(canvasElement)
      const theElements = canvas.getAllByRole('button')
      // ðŸ‘‡ Simulate interactions with the component
      await expect(theElements).not.toBeNull()
      await expect(theElements.length).toBe(3)
      await expect(theElements[0].textContent).toEqual('Button 1')
      await expect(theElements[1].textContent).toEqual('Button 2')
      await expect(theElements[2].textContent).toEqual('Button 3')
    }}
  >
    {Template.bind({})}
  </Story>
</Canvas>
